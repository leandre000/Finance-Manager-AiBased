STEP 1: CREATE DATABASE
━━━━━━━━━━━━━━━━━━━━━━

1. Click: "New +" button
2. Select: "PostgreSQL"
3. Fill in:
   Name: finance-manager-db
   Database: finance_manager
   Region: Oregon
   Plan: Free ✓
4. Click: "Create Database"

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 2: COPY DATABASE URL
━━━━━━━━━━━━━━━━━━━━━━━━

1. Wait for "Available" status (2 min)
2. Click on "finance-manager-db"
3. Scroll to "Connections"
4. Copy "Internal Database URL"
   
   Example:
   postgresql://finance_user:abc123@dpg-xxxxx-a/finance_manager

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 3: SET IN WEB SERVICE
━━━━━━━━━━━━━━━━━━━━━━━━━

1. Dashboard → "finance-manager-api"
2. Click "Environment" tab
3. Find DATABASE_URL
4. Click "Edit"
5. Paste the URL you copied
6. Click "Save Changes"

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Auto-deploys when you save!
OR: Manual Deploy → Deploy latest commit
